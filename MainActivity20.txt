package com.example.myapplication;

import androidx.appcompat.app.AppCompatActivity;

import android.content.Intent;
import android.os.Bundle;
import android.widget.EditText;
import android.widget.TextView;

public class MainActivity20 extends AppCompatActivity {
    TextView UN;
    TextView Pno;
    TextView ADD;
    TextView pin;
    TextView pay;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main20);
        UN =(TextView)findViewById(R.id.Uname);
        Intent i = getIntent();
        String st = i.getStringExtra("Message");
        UN.setText(st);
        Pno =(TextView)findViewById(R.id.Pno);
        String st1 = i.getStringExtra("Message1");
        Pno.setText(st1);
        ADD =(TextView)findViewById(R.id.Add);
        String st2 = i.getStringExtra("Message2");
        ADD.setText(st2);
        pin =(TextView)findViewById(R.id.pin);
        String st3 = i.getStringExtra("Message3");
        pin.setText(st3);
        pay =(TextView)findViewById(R.id.pay);
        String st4 = i.getStringExtra("Message4");
        pay.setText(st4);
        EditText ed= (EditText)findViewById(R.id.et);
        ed.setText("Payment received successfully");
    }
}
